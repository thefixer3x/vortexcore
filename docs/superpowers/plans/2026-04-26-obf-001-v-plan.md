**Goal:** Add a Providus OBF accounts panel to the VortexCore dashboard, gated by `VITE_OBF_LIVE`, that calls the `obf-accounts` edge function and renders real bank data — without touching the existing internal wallet/transaction display.

**Architecture:** New Supabase Edge Function (`obf-accounts`) proxies requests to `onasis-gateway`. New client service + React hook call it from the browser. A new `OBFAccountPanel` component renders only when the feature flag is `true`. Existing dashboard code is untouched.

**Tech Stack:** Deno (edge function), React Query + custom hook (client), zod (validation), `@tanstack/react-query` (already in project)

```ts
// supabase/functions/obf-accounts/index.ts
import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2.39.0";
import { z } from "https://deno.land/x/zod@v0.2.2/mod.ts";

const GATEWAY_URL = Deno.env.get("ONASIS_GATEWAY_URL");
const GATEWAY_TOKEN = Deno.env.get("ONASIS_GATEWAY_TOKEN");

const AccountSchema = z.object({
  id: z.string(),
  displayName: z.string(),
  maskedNumber: z.string(),
  type: z.string(),
  currency: z.string(),
  balance: z.number(),
  asOf: z.string(),
});

const TransactionSchema = z.object({
  id: z.string(),
  accountId: z.string(),
  postedAt: z.string(),
  amount: z.number(),
  currency: z.string(),
  direction: z.enum(["credit", "debit"]),
  counterparty: z.string(),
  description: z.string(),
  status: z.enum(["posted", "pending", "failed"]),
});

const ErrorSchema = z.object({
  code: z.string(),
  message: z.string(),
  upstreamStatus: z.number().optional(),
});

async function withAuth(req: Request) {
  const authHeader = req.headers.get("Authorization");
  if (!authHeader) {
    return new Response(JSON.stringify({ error: "Unauthorized" }), {
      status: 401,
      headers: { "Content-Type": "application/json" },
    });
  }
  const supabaseUrl = Deno.env.get("SUPABASE_URL")!;
  const supabaseServiceKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;
  const supabase = createClient(supabaseUrl, supabaseServiceKey);
  const token = authHeader.replace("Bearer ", "");
  const { data: { user }, error } = await supabase.auth.getUser(token);
  if (error || !user) {
    return new Response(JSON.stringify({ error: "Invalid session" }), {
      status: 401,
      headers: { "Content-Type": "application/json" },
    });
  }
  return { userId: user.id, authHeader };
}

async function proxyToGateway(path: string, req?: Request) {
  if (!GATEWAY_URL || !GATEWAY_TOKEN) {
    return new Response(JSON.stringify({
      code: "gateway_not_configured",
      message: "onasis-gateway is not configured on this environment.",
    }), {
      status: 503,
      headers: { "Content-Type": "application/json" },
    });
  }
  const url = `${GATEWAY_URL}${path}`;
  const upstreamResp = await fetch(url, {
    headers: {
      "Authorization": `Bearer ${GATEWAY_TOKEN}`,
      "Content-Type": "application/json",
    },
    ...(req ? { method: req.method, body: req.body } : {}),
  });
  if (!upstreamResp.ok) {
    let body;
    try { body = ErrorSchema.parse(await upstreamResp.json()); }
    catch { body = { code: "gateway_error", message: upstreamResp.statusText }; }
    return new Response(JSON.stringify(body), {
      status: upstreamResp.status,
      headers: { "Content-Type": "application/json" },
    });
  }
  return new Response(upstreamResp.body, {
    status: upstreamResp.status,
    headers: { "Content-Type": "application/json" },
  });
}

serve(async (req) => {
  const auth = await withAuth(req);
  if (auth instanceof Response) return auth;

  const url = new URL(req.url);
  const pathname = url.pathname;

  if (req.method === "GET" && pathname === "/obf-accounts/") {
    const gatewayResp = await proxyToGateway("/v1/aisp/accounts", req);
    const body = await gatewayResp.json();
    const parsed = z.array(AccountSchema).safeParse(body);
    if (!parsed.success) {
      return new Response(JSON.stringify({
        code: "schema_violation",
        message: "Gateway returned an unexpected account shape.",
        upstreamStatus: 502,
      }), { status: 502, headers: { "Content-Type": "application/json" } });
    }
    return gatewayResp;
  }

  const txMatch = pathname.match(/^\/obf-accounts\/([^/]+)\/transactions$/);
  if (req.method === "GET" && txMatch) {
    const accountId = txMatch[1];
    const from = url.searchParams.get("from") ?? "";
    const to = url.searchParams.get("to") ?? "";
    const limit = url.searchParams.get("limit") ?? "50";
    const params = new URLSearchParams();
    if (from) params.set("from", from);
    if (to) params.set("to", to);
    params.set("limit", limit);
    const query = params.toString();
    const gatewayResp = await proxyToGateway(
      `/v1/aisp/accounts/${accountId}/transactions${query ? `?${query}` : ""}`, req
    );
    const body = await gatewayResp.json();
    const parsed = z.array(TransactionSchema).safeParse(body);
    if (!parsed.success) {
      return new Response(JSON.stringify({
        code: "schema_violation",
        message: "Gateway returned an unexpected transaction shape.",
        upstreamStatus: 502,
      }), { status: 502, headers: { "Content-Type": "application/json" } });
    }
    return gatewayResp;
  }

  return new Response(JSON.stringify({ error: "Not found" }), {
    status: 404,
    headers: { "Content-Type": "application/json" },
  });
});
```

```ts
// src/services/obf/accounts.ts
import { supabase } from "@/integrations/supabase/client";

export interface OBFAccount {
  id: string;
  displayName: string;
  maskedNumber: string;
  type: string;
  currency: string;
  balance: number;
  asOf: string;
}

export interface OBFTransaction {
  id: string;
  accountId: string;
  postedAt: string;
  amount: number;
  currency: string;
  direction: "credit" | "debit";
  counterparty: string;
  description: string;
  status: "posted" | "pending" | "failed";
}

export interface OBFFetchError {
  code: string;
  message: string;
  upstreamStatus?: number;
}

const isOBFError = (body: unknown): body is OBFFetchError =>
  typeof body === "object" && body !== null && "code" in body && "message" in body;

export async function fetchOBFAccounts(): Promise<OBFAccount[]> {
  const { data: { session } } = await supabase.auth.getSession();
  if (!session) throw new Error("Not authenticated");

  const { data, error: invokeError } = await supabase.functions.invoke("obf-accounts", {
    method: "GET",
    headers: { Authorization: `Bearer ${session.access_token}` },
  });

  if (invokeError) throw new Error(invokeError.message ?? "Failed to invoke obf-accounts");
  if (!isOBFError(data) && !Array.isArray(data)) throw new Error("Unexpected response shape");
  if (isOBFError(data)) {
    if (data.code === "gateway_not_configured") throw new Error("Live Providus data is not available.");
    throw new Error(data.message ?? `Gateway error: ${data.code}`);
  }
  return data as OBFAccount[];
}

export async function fetchOBFTransactions(
  accountId: string,
  opts?: { from?: string; to?: string; limit?: number }
): Promise<OBFTransaction[]> {
  const { data: { session } } = await supabase.auth.getSession();
  if (!session) throw new Error("Not authenticated");

  const params = new URLSearchParams();
  if (opts?.from) params.set("from", opts.from);
  if (opts?.to) params.set("to", opts.to);
  if (opts?.limit) params.set("limit", String(opts.limit));
  const query = params.toString();

  const { data, error: invokeError } = await supabase.functions.invoke(
    `obf-accounts/${accountId}/transactions${query ? `?${query}` : ""}`,
    { method: "GET", headers: { Authorization: `Bearer ${session.access_token}` } }
  );

  if (invokeError) throw new Error(invokeError.message ?? "Failed to invoke obf-accounts");
  if (!isOBFError(data) && !Array.isArray(data)) throw new Error("Unexpected response shape");
  if (isOBFError(data)) {
    if (data.code === "gateway_not_configured") throw new Error("Live Providus data is not available.");
    throw new Error(data.message ?? `Gateway error: ${data.code}`);
  }
  return data as OBFTransaction[];
}
```

```ts
// src/hooks/use-obf-accounts.ts
import { useQuery } from "@tanstack/react-query";
import { fetchOBFAccounts, fetchOBFTransactions, type OBFAccount, type OBFTransaction } from "@/services/obf/accounts";

const OBF_LIVE = import.meta.env.VITE_OBF_LIVE === "true";

export function useOBFAccounts() {
  return useQuery<OBFAccount[], Error>({
    queryKey: ["obf", "accounts"],
    queryFn: fetchOBFAccounts,
    enabled: OBF_LIVE,
    staleTime: 1000 * 60 * 2,
    retry: 1,
  });
}

export function useOBFTransactions(accountId: string | null) {
  return useQuery<OBFTransaction[], Error>({
    queryKey: ["obf", "transactions", accountId],
    queryFn: () => fetchOBFTransactions(accountId!),
    enabled: OBF_LIVE && !!accountId,
    staleTime: 1000 * 60 * 2,
    retry: 1,
  });
}
```

```tsx
// src/components/obf/OBFAccountPanel.tsx
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";
import { RefreshCw, AlertCircle, ChevronRight } from "lucide-react";
import { OBFAccountCard } from "./OBFAccountCard";
import { OBFTransactionList } from "./OBFTransactionList";
import { Skeleton } from "@/components/ui/skeleton";
import { useOBFAccounts, useOBFTransactions } from "@/hooks/use-obf-accounts";
import { useState } from "react";

export const OBFAccountPanel = () => {
  const { data: accounts, isLoading, error, refetch } = useOBFAccounts();
  const [selectedAccountId, setSelectedAccountId] = useState<string | null>(
    accounts?.[0]?.id ?? null
  );
  const { data: transactions } = useOBFTransactions(selectedAccountId);

  const displayAccount = accounts?.find((a) => a.id === selectedAccountId) ?? accounts?.[0] ?? null;

  return (
    <Card className="border-0 shadow-lg">
      <CardHeader className="pb-4">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <CardTitle className="text-xl font-bold">OBF Accounts</CardTitle>
            <span className="text-xs text-muted-foreground">(Providus via onasis-gateway)</span>
          </div>
          {error && (
            <Button variant="outline" size="sm" onClick={() => void refetch()} className="gap-1">
              <RefreshCw className="h-3 w-3" />
              Retry
            </Button>
          )}
        </div>
      </CardHeader>

      <CardContent className="space-y-4">
        {error && (
          <Alert variant="destructive">
            <AlertCircle className="h-4 w-4" />
            <AlertTitle>Unable to load OBF accounts</AlertTitle>
            <AlertDescription>{error.message}</AlertDescription>
          </Alert>
        )}

        {isLoading && !accounts && (
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
            {Array.from({ length: 3 }).map((_, i) => (
              <Skeleton key={i} className="h-[160px] rounded-xl" />
            ))}
          </div>
        )}

        {accounts && accounts.length > 0 && (
          <>
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
              {accounts.map((account) => (
                <div
                  key={account.id}
                  onClick={() => setSelectedAccountId(account.id)}
                  className="cursor-pointer"
                >
                  <OBFAccountCard
                    account={account}
                    className={
                      selectedAccountId === account.id
                        ? "ring-2 ring-primary ring-offset-2"
                        : ""
                    }
                  />
                </div>
              ))}
            </div>

            {displayAccount && (
              <div className="mt-4">
                <div className="flex items-center justify-between mb-2">
                  <h4 className="text-sm font-medium">
                    Transactions — {displayAccount.displayName}
                  </h4>
                  <ChevronRight className="h-4 w-4 text-muted-foreground" />
                </div>
                <OBFTransactionList transactions={transactions ?? []} />
              </div>
            )}
          </>
        )}

        {!isLoading && !error && accounts?.length === 0 && (
          <div className="py-8 text-center text-muted-foreground text-sm">
            No OBF accounts found for this user.
          </div>
        )}
      </CardContent>
    </Card>
  );
};
```
